42  import java.net.HttpURLConnection;
43  import java.net.URI;

46  import javax.json.JsonObject;
47  import javax.json.JsonValue;

54  import org.junit.jupiter.api.Test;

59  /**
60   * Tests for any implementation of {@link Request}.
61   *
62   * @checkstyle JavadocMethodCheck (500 lines)
63   * @since 1.17.8
64   */
65  @RequiredArgsConstructor(access = AccessLevel.PROTECTED)
66  @SuppressWarnings({"PMD.AbstractClassWithoutAbstractMethod", "PMD.TooManyMethods"})
67  public abstract class RequestITCaseTemplate {
68  
69      /**
70       * Type of Request.
71       */
72      private final Class<? extends Request> type;
73  
74      /**
75       * Base URI.
76       */
77      private final URI uri;
78  
79      /**
80       * Make request for a specific path.
81       * @param path Path.
82       * @return Request.
83       */
84      protected final Request request(final String path) {
85          return RequestTestTemplate.request(
86              this.uri.resolve(path),
87              this.type
88          );
89      }
90  
91      @ParameterizedTest
92      @ValueSource(ints = {
93          HttpURLConnection.HTTP_NOT_FOUND,
94          HttpURLConnection.HTTP_OK,
95          HttpURLConnection.HTTP_UNAVAILABLE
96      })
97      final void readsReturnStatusCode(final int code) throws IOException {
98          this.request(String.format("/status/%d", code))
99              .fetch()
100             .as(RestResponse.class)
101             .assertStatus(code);
102     }
103 
104     @ParameterizedTest
105     @ValueSource(
106         strings = {
107             Request.DELETE,
108             Request.GET,
109             Request.POST,
110             Request.PUT
111         }
112     )
113     final void readsReturnStatusCode(final String method) throws IOException {
114         this.request(String.format("/%s", method.toLowerCase(Locale.ENGLISH)))
115             .method(method)
116             .fetch()
117             .as(RestResponse.class)
118             .assertStatus(HttpURLConnection.HTTP_OK);
119     }
120 
121     @Test
122     final void sendsCookies() throws IOException {
123         MatcherAssert.assertThat(
124             "Must send cookies",
125             this.request("/cookies")
126                 .header(HttpHeaders.COOKIE, "foo=bar")
127                 .header(HttpHeaders.COOKIE, "baz=foz")
128                 .through(CookieOptimizingWire.class)
129                 .fetch()
130                 .as(JsonResponse.class)
131                 .json()
132                 .readObject()
133                 .getJsonObject("cookies"),
134             Matchers.allOf(
135                 Matchers.hasEntry(
136                     Matchers.is("foo"),
137                     Matchers.is(Json.createValue("bar"))
138                 ),
139                 Matchers.hasEntry(
140                     Matchers.is("baz"),
141                     Matchers.is(Json.createValue("foz"))
142                 )
143             )
144         );
145     }
146 
147     @Test
148     final void followsLocationHeader() throws IOException {
149         this.request("/absolute-redirect/5")
150             .through(AutoRedirectingWire.class, Tv.SIX)
151             .fetch()
152             .as(RestResponse.class)
153             .assertStatus(HttpURLConnection.HTTP_OK);
154     }
155 
156     @Test
157     final void followsLocationHeaderRelativeRedirect() throws IOException {
158         this.request("/relative-redirect/5")
159             .through(AutoRedirectingWire.class, Tv.SIX)
160             .fetch()
161             .as(RestResponse.class)
162             .assertStatus(HttpURLConnection.HTTP_OK);
163     }
164 
165     @Test
166     final void failsOnTimeout() {
167         Assertions.assertThrows(
168             IOException.class,
169             () ->
170                 this.request("/delay/3")
171                     .timeout(Tv.THOUSAND, Tv.THOUSAND)
172                     .fetch()
173         );
174     }
175 
176     @Test
177     final void readsJsonResponse() throws Exception {
178         MatcherAssert.assertThat(
179             "Must parse Json response",
180             this.request("/json")
181                 .fetch()
182                 .as(JsonResponse.class)
183                 .json()
184                 .readObject(),
185             Matchers.notNullValue(JsonObject.class)
186         );
187     }
188 
189     @Test
190     @DisabledIf("isJdkRequest")
191     final void readsDeflatedJsonResponse() throws Exception {
192         MatcherAssert.assertThat(
193             "Must undeflate & parse Json response",
194             this.request("/deflate")
195                 .fetch()
196                 .as(JsonResponse.class)
197                 .json()
198                 .readObject(),
199             Matchers.hasEntry(
200                 Matchers.is("deflated"),
201                 Matchers.is(JsonValue.TRUE)
202             )
203         );
204     }
205 
206     @Test
207     @DisabledIf("isJdkRequest")
208     final void readsGzippedJsonResponse() throws Exception {
209         MatcherAssert.assertThat(
210             "Must unzip & parse Json response",
211             this.request("/gzip")
212                 .fetch()
213                 .as(JsonResponse.class)
214                 .json()
215                 .readObject(),
216             Matchers.hasEntry(
217                 Matchers.is("gzipped"),
218                 Matchers.is(JsonValue.TRUE)
219             )
220         );
221     }
222 
223     @Test
224     final void handlesBasicAuth() throws Exception {
225         MatcherAssert.assertThat(
226             "Must authenticate with userInfo",
227             this.request("/basic-auth/jeff/secret")
228                 .uri()
229                 .userInfo("jeff:secret")
230                 .back()
231                 .through(BasicAuthWire.class)
232                 .fetch()
233                 .as(JsonResponse.class)
234                 .status(),
235             Matchers.is(
236                 HttpURLConnection.HTTP_OK
237             )
238         );
239     }
240 
241     @Test
242     final void readsXmlResponse() throws Exception {
243         MatcherAssert.assertThat(
244             "Must parse XML response",
245             this.request("/xml")
246                 .fetch()
247                 .as(XmlResponse.class)
248                 .xml(),
249             Matchers.notNullValue(XML.class)
250         );
251     }
252 
253     /**
254      * Is JdkRequest being tested?
255      * @return True if so.
256      */
257     @SuppressWarnings("PMD.UnusedPrivateMethod")
258     private boolean isJdkRequest() {
259         return JdkRequest.class.equals(this.type);
260     }
261 }
