218             new LinkedList<>();
219         final String key = ImmutableHeader.normalize(name);
220         for (final Map.Entry<String, String> header : this.hdrs) {
221             if (!header.getKey().equals(key)) {
222                 headers.add(header);
223             }
224         }
225         return new BaseRequest(
226             this.wire,
227             this.home,
228             headers,
229             this.mtd,
230             this.content,
231             this.connect,
232             this.read
233         );

308     @Override
309     @SuppressWarnings("PMD.ConsecutiveLiteralAppends")
310     public String toString() {
311         final URI uri = URI.create(this.home);
312         final StringBuilder text = new StringBuilder("HTTP/1.1 ")
313             .append(this.mtd).append(' ')
314             .append(uri.getPath())
315             .append(" (")
316             .append(uri.getHost())
317             .append(")\n");
318         for (final Map.Entry<String, String> header : this.hdrs) {
319             text.append(
320                 Logger.format(
321                     "%s: %s\n",
322                     header.getKey(),
323                     header.getValue()
324                 )
325             );
326         }
327         return text.append('\n')
328             .append(new RequestBody.Printable(this.content).toString())
329             .toString();
330     }

332     /**
333      * Create an instance of Wire.
334      *
335      * @param type Type of Wire.
336      * @param args Ctor arguments.
337      * @param <T> Type of Wire.
338      * @return An instance of Wire
339      */
340     private <T extends Wire> Wire mkWire(
341         final Class<T> type,
342         final Object... args
343     ) {
344         final Constructor<?> ctor = BaseRequest.findCtor(type, args);
345         final Object[] params = new Object[args.length + 1];
346         params[0] = this.wire;
347         System.arraycopy(args, 0, params, 1, args.length);
348         final Wire decorated;
349         try {
350             decorated = Wire.class.cast(ctor.newInstance(params));
351         } catch (final InstantiationException
352             | IllegalAccessException | InvocationTargetException ex) {
353             throw new IllegalStateException(ex);
354         }
355         return decorated;
356     }

358     /**
359      * Fetch response from server.
360      * @param stream The content to send.
361      * @return The obtained response
362      * @throws IOException If an IO exception occurs.
363      */
364     private Response fetchResponse(final InputStream stream)
365         throws IOException {
366         final long start = System.currentTimeMillis();
367         final Response response = this.wire.send(
368             this, this.home, this.mtd,
369             this.hdrs, stream, this.connect,
370             this.read
371         );
372         final URI uri = URI.create(this.home);
373         if (Logger.isInfoEnabled(this)) {
374             Logger.info(
375                 this,
376                 "#fetch(%s %s%s %s): [%d %s] in %[ms]s",
377                 this.mtd,
378                 uri.getHost(),
379                 // @checkstyle AvoidInlineConditionalsCheck (1 line)
380                 uri.getPort() > 0 ? String.format(":%d", uri.getPort()) : "",
381                 uri.getPath(),
382                 response.status(),
383                 response.reason(),
384                 System.currentTimeMillis() - start
385             );
386         }
387         return response;
388     }

390     /**
391      * Create uri from String.
392      * @param uri String
393      * @return URI
394      */
395     private static URI createUri(final String uri) {
396         URI addr = URI.create(uri);
397         if (addr.getPath() != null && addr.getPath().isEmpty()) {
398             addr = UriBuilder.fromUri(addr).path("/").build();
399         }
400         return addr;
401     }

403     /**
404      * Find a ctor which match arguments.
405      * @param type A type.
406      * @param args Ctor arguments.
407      * @param <T> Type of object
408      * @return A proper ctor for args.
409      */
410     private static <T extends Wire> Constructor<?> findCtor(
411         final Class<T> type, final Object... args
412     ) {
413         Constructor<?> ctor = null;
414         for (final Constructor<?> opt : type.getDeclaredConstructors()) {
415             final Class<?>[] types = opt.getParameterTypes();
416             if (types.length == args.length + 1) {
417                 boolean match = true;
418                 for (int inx = 1; inx < types.length && match; ++inx) {
419                     match = BaseRequest
420                         .wrappedIfNeeded(types[inx])
421                         .isAssignableFrom(args[inx - 1].getClass());
422                 }
423                 if (match) {
424                     ctor = opt;
425                     break;
426                 }
427             }
428         }
429         if (ctor == null) {
430             throw new IllegalArgumentException(
431                 String.format(
432                     "class %s doesn't have a ctor with %d argument(s)",
433                     type.getName(), args.length
434                 )
435             );
436         }
437         return ctor;
438     }

440     /**
441      * Wrap primitive types.
442      * @param type A type which could be primitive
443      * @return Wrapped type if it was a primitive
444      */
445     private static Class<?> wrappedIfNeeded(final Class<?> type) {
446         Class<?> arg = type;
447         if (arg.isPrimitive()) {
448             arg = ClassUtil.wrapperType(arg);
449         }
450         return arg;
451     }

533             );

633         @Override
634         public RequestBody formParam(final String name, final Object value) {
635             final String boundary = this.boundary();
636             final String dashes = "--";
637             final byte[] last = Arrays.copyOfRange(
638                 this.text,
639                 Math.max(this.text.length - 2, 0),
640                 this.text.length
641             );
642             final byte[] old;
643             if (Arrays.equals(last, dashes.getBytes(BaseRequest.CHARSET))) {
644                 old = Arrays.copyOf(this.text, this.text.length - 2);
645             } else {
646                 old = String.format("%s%s", dashes, boundary)
647                     .getBytes(BaseRequest.CHARSET);
648             }
649             final byte[] bytes;
650             if (value instanceof byte[]) {
651                 bytes = (byte[]) value;
652             } else {
653                 bytes = value.toString().getBytes(BaseRequest.CHARSET);
654             }
655             final byte[] disposition = Joiner.on("; ")
656                 .join(
657                     "Content-Disposition: form-data",
658                     String.format("name=\"%s\"", name),
659                     "filename=\"binary\""
660                 ).getBytes(BaseRequest.CHARSET);
661             final byte[] type = "Content-Type: application/octet-stream"
662                 .getBytes(BaseRequest.CHARSET);
663             final byte[] footer = String.format(
664                 "%s%s%s", dashes, boundary, dashes
665             ).getBytes(BaseRequest.CHARSET);
666             final MultipartBodyBuilder neww = new MultipartBodyBuilder()
667                 .appendLine(old)
668                 .appendLine(disposition)
669                 .appendLine(type)
670                 .appendLine(new byte[0])
671                 .appendLine(bytes)
672                 .append(footer);
673             return new BaseRequest.MultipartFormBody(
674                 this.owner,
675                 neww.asBytes()
676             );
677         }
678 
679         @Override
680         public RequestBody formParams(final Map<String, String> params) {
681             RequestBody body = this;
682             for (final Map.Entry<String, String> param : params.entrySet()) {
683                 body = body.formParam(param.getKey(), param.getValue());
684             }
685             return body;
686         }
687 
688         /**
689          * Boundary value found.
690          * @return Boundary string.
691          */
692         private String boundary() {
693             for (final Map.Entry<String, String> hdr : this.owner.hdrs) {
694                 if (hdr.getKey().equals(HttpHeaders.CONTENT_TYPE)
695                     && hdr.getValue().matches(".*;\\s*[bB]oundary=.*")) {
696                     return hdr.getValue()
697                         .replaceFirst(".*;\\s*[bB]oundary=", "");
698                 }
699             }
700             throw new IllegalStateException(
701                 "Content-Type: multipart/form-data requires boundary"
702             );
703         }
704     }

780         @Override
781         public RequestBody formParam(final String name, final Object value) {
782             try {
783                 final StringBuilder builder = new StringBuilder(this.get());
784                 if (!builder.toString().isEmpty()) {
785                     builder.append('&');
786                 }
787                 return new BaseRequest.FormEncodedBody(
788                     this.owner,
789                     builder
790                         .append(name)
791                         .append('=')
792                         .append(
793                             URLEncoder.encode(
794                                 value.toString(),
795                                 BaseRequest.ENCODING
796                             )
797                         )
798                         .toString()
799                         .getBytes(BaseRequest.CHARSET)
800                 );
801             } catch (final UnsupportedEncodingException ex) {
802                 throw new IllegalStateException(ex);
803             }
804         }
