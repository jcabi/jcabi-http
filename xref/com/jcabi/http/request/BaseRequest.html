165     /**
166      * Public ctor.
167      * @param wre Wire
168      * @param uri The resource to work with
169      * @param headers Headers
170      * @param method HTTP method
171      * @param body HTTP request body
172      * @param cnct Connect timeout for http connection
173      * @param rdd Read timeout for http connection
174      * @checkstyle ParameterNumber (5 lines)
175      */
176     BaseRequest(final Wire wre, final String uri,
177         final Iterable<Map.Entry<String, String>> headers,
178         final String method, final byte[] body,
179         final int cnct, final int rdd) {
180         this.wire = wre;
181         URI addr = URI.create(uri);
182         if (addr.getPath() != null && addr.getPath().isEmpty()) {
183             addr = UriBuilder.fromUri(addr).path("/").build();
184         }
185         this.home = addr.toString();
186         this.hdrs = new Array<Map.Entry<String, String>>(headers);
187         this.mtd = method;
188         this.content = body.clone();
189         this.connect = cnct;
190         this.read = rdd;
191     }

211     @Override
212     public Request reset(final String name) {
213         final Collection<Map.Entry<String, String>> headers =
214             new LinkedList<>();
215         final String key = ImmutableHeader.normalize(name);
216         for (final Map.Entry<String, String> header : this.hdrs) {
217             if (!header.getKey().equals(key)) {
218                 headers.add(header);
219             }
220         }
221         return new BaseRequest(
222             this.wire,
223             this.home,
224             headers,
225             this.mtd,
226             this.content,
227             this.connect,
228             this.read
229         );
230     }

273     @Override
274     public Response fetch(final InputStream stream) throws IOException {
275         if (this.content.length > 0) {
276             throw new IllegalStateException(
277                 "Request Body is not empty, use fetch() instead"
278             );
279         }
280         return this.fetchResponse(stream);
281     }

283     @Override
284     public <T extends Wire> Request through(final Class<T> type,
285         final Object... args) {
286         Constructor<?> ctor = null;
287         for (final Constructor<?> opt : type.getDeclaredConstructors()) {
288             if (opt.getParameterTypes().length == args.length + 1) {
289                 ctor = opt;
290                 break;
291             }
292         }
293         if (ctor == null) {
294             throw new IllegalArgumentException(
295                 String.format(
296                     "class %s doesn't have a ctor with %d argument(s)",
297                     type.getName(), args.length
298                 )
299             );
300         }
301         final Object[] params = new Object[args.length + 1];
302         params[0] = this.wire;
303         System.arraycopy(args, 0, params, 1, args.length);
304         final Wire decorated;
305         try {
306             decorated = Wire.class.cast(ctor.newInstance(params));
307         } catch (final InstantiationException
308             | IllegalAccessException | InvocationTargetException ex) {
309             throw new IllegalStateException(ex);
310         }
311         return new BaseRequest(
312             decorated,
313             this.home,
314             this.hdrs,
315             this.mtd,
316             this.content,
317             this.connect,
318             this.read
319         );
320     }

322     @Override
323     @SuppressWarnings("PMD.ConsecutiveLiteralAppends")
324     public String toString() {
325         final URI uri = URI.create(this.home);
326         final StringBuilder text = new StringBuilder("HTTP/1.1 ")
327             .append(this.mtd).append(' ')
328             .append(uri.getPath())
329             .append(" (")
330             .append(uri.getHost())
331             .append(")\n");
332         for (final Map.Entry<String, String> header : this.hdrs) {
333             text.append(
334                 Logger.format(
335                     "%s: %s\n",
336                     header.getKey(),
337                     header.getValue()
338                 )
339             );
340         }
341         return text.append('\n')
342             .append(new RequestBody.Printable(this.content).toString())
343             .toString();
344     }

346     /**
347      * Fetch response from server.
348      * @param stream The content to send.
349      * @return The obtained response
350      * @throws IOException If an IO exception occurs.
351      */
352     private Response fetchResponse(final InputStream stream)
353         throws IOException {
354         final long start = System.currentTimeMillis();
355         final Response response = this.wire.send(
356             this, this.home, this.mtd,
357             this.hdrs, stream, this.connect,
358             this.read
359         );
360         final URI uri = URI.create(this.home);
361         Logger.info(
362             this,
363             "#fetch(%s %s%s %s): [%d %s] in %[ms]s",
364             this.mtd,
365             uri.getHost(),
366             // @checkstyle AvoidInlineConditionalsCheck (1 line)
367             uri.getPort() > 0 ? String.format(":%d", uri.getPort()) : "",
368             uri.getPath(),
369             response.status(),
370             response.reason(),
371             System.currentTimeMillis() - start
372         );
373         return response;
374     }

376     /**
377      * Base URI.
378      */
379     @Immutable
380     @EqualsAndHashCode(of = "address")
381     @Loggable(Loggable.DEBUG)
382     private static final class BaseURI implements RequestURI {
383         /**
384          * URI encapsulated.
385          */
386         private final transient String address;
387         /**
388          * Base request encapsulated.
389          */
390         private final transient BaseRequest owner;
391         /**
392          * Public ctor.
393          * @param req Request
394          * @param uri The URI to start with
395          */
396         BaseURI(final BaseRequest req, final String uri) {
397             this.owner = req;
398             this.address = uri;
399         }
400         @Override
401         public String toString() {
402             return this.address;
403         }
404         @Override
405         public Request back() {
406             return new BaseRequest(
407                 this.owner.wire,
408                 this.address,
409                 this.owner.hdrs,
410                 this.owner.mtd,
411                 this.owner.content
412             );
413         }
414         @Override
415         public URI get() {
416             return URI.create(this.owner.home);
417         }
418         @Override
419         public RequestURI set(final URI uri) {
420             return new BaseRequest.BaseURI(this.owner, uri.toString());
421         }
422         @Override
423         public RequestURI queryParam(final String name, final Object value) {
424             return new BaseRequest.BaseURI(
425                 this.owner,
426                 UriBuilder.fromUri(this.address)
427                     .queryParam(name, "{value}")
428                     .build(value).toString()
429             );
430         }
431         @Override
432         public RequestURI queryParams(final Map<String, String> map) {
433             final UriBuilder uri = UriBuilder.fromUri(this.address);
434             final Object[] values = new Object[map.size()];
435             int idx = 0;
436             for (final Map.Entry<String, String> pair : map.entrySet()) {
437                 uri.queryParam(pair.getKey(), String.format("{x%d}", idx));
438                 values[idx] = pair.getValue();
439                 ++idx;
440             }
441             return new BaseRequest.BaseURI(
442                 this.owner,
443                 uri.build(values).toString()
444             );
445         }
446         @Override
447         public RequestURI path(final String segment) {
448             return new BaseRequest.BaseURI(
449                 this.owner,
450                 UriBuilder.fromUri(this.address)
451                     .path(segment)
452                     .build().toString()
453             );
454         }
455         @Override
456         public RequestURI userInfo(final String info) {
457             return new BaseRequest.BaseURI(
458                 this.owner,
459                 UriBuilder.fromUri(this.address)
460                     .userInfo(info)
461                     .build().toString()
462             );
463         }
464         @Override
465         public RequestURI port(final int num) {
466             return new BaseRequest.BaseURI(
467                 this.owner,
468                 UriBuilder.fromUri(this.address)
469                     .port(num).build().toString()
470             );
471         }
472     }

535         @Override

562     /**
563      * Body of a request with a simple form.
564      * (enctype application/x-www-form-urlencoded)
565      */
566     @Immutable
567     @EqualsAndHashCode(of = "text")
568     @Loggable(Loggable.DEBUG)
569     private static final class FormEncodedBody implements RequestBody {
570 
571         /**
572          * Content encapsulated.
573          */
574         @Immutable.Array
575         private final transient byte[] text;
576 
577         /**
578          * Base request encapsulated.
579          */
580         private final transient BaseRequest owner;
581 
582         /**
583          * URL form character to prepend.
584          */
585         private final transient String prepend;
586 
587         /**
588          * Public ctor.
589          * @param req Request
590          * @param body Text to encapsulate
591          */
592         FormEncodedBody(final BaseRequest req, final byte[] body) {
593             this(req, body, "");
594         }
595 
596         /**
597          * Public ctor.
598          * @param req Request
599          * @param body Text to encapsulate
600          * @param pre Character to prepend
601          */
602         FormEncodedBody(
603             final BaseRequest req, final byte[] body, final String pre
604         ) {
605             this.owner = req;
606             this.text = body.clone();
607             this.prepend = pre;
608         }
609 
610         @Override
611         public String toString() {
612             return new RequestBody.Printable(this.text).toString();
613         }
614 
615         @Override
616         public Request back() {
617             return new BaseRequest(
618                 this.owner.wire,
619                 this.owner.home,
620                 this.owner.hdrs,
621                 this.owner.mtd,
622                 this.text,
623                 this.owner.connect,
624                 this.owner.read
625             );
626         }
627 
628         @Override
629         public String get() {
630             return new String(this.text, BaseRequest.CHARSET);
631         }
632 
633         @Override
634         public RequestBody set(final String txt) {
635             return this.set(txt.getBytes(BaseRequest.CHARSET));
636         }
637 
638         @Override
639         public RequestBody set(final JsonStructure json) {
640             final StringWriter writer = new StringWriter();
641             Json.createWriter(writer).write(json);
642             return this.set(writer.toString());
643         }
644 
645         @Override
646         public RequestBody set(final byte[] txt) {
647             return new BaseRequest.FormEncodedBody(this.owner, txt);
648         }
649 
650         @Override
651         public RequestBody formParam(final String name, final Object value) {
652             try {
653                 return new BaseRequest.FormEncodedBody(
654                     this.owner,
655                     new StringBuilder(this.get())
656                         .append(this.prepend)
657                         .append(name)
658                         .append('=')
659                         .append(
660                             URLEncoder.encode(
661                                 value.toString(),
662                                 BaseRequest.ENCODING
663                             )
664                         )
665                         .toString()
666                         .getBytes(BaseRequest.CHARSET),
667                     "&"
668                 );
669             } catch (final UnsupportedEncodingException ex) {
670                 throw new IllegalStateException(ex);
671             }
672         }
673 
674         @Override
675         public RequestBody formParams(final Map<String, String> params) {
676             RequestBody body = this;
677             for (final Map.Entry<String, String> param : params.entrySet()) {
678                 body = body.formParam(param.getKey(), param.getValue());
679             }
680             return body;
681         }
