30  package com.jcabi.http.mock;
31  
32  import org.hamcrest.Matcher;
33  
34  /**
35   * Convenient set of matchers for {@link MkAnswer}.
36   * @since 1.5
37   */
38  @SuppressWarnings("PMD.ProhibitPublicStaticMethods")
39  public final class MkAnswerMatchers {
40      /**
41       * Private ctor.
42       */
43      private MkAnswerMatchers() {
44          // Utility class - cannot instantiate.
45      }
46  
47      /**
48       * Matches the value of the MkAnswer's body against the given matcher.
49       *
50       * @param matcher The matcher to use.
51       * @return Matcher for checking the body of MkAnswer
52       */
53      public static Matcher<MkAnswer> hasBody(final Matcher<String> matcher) {
54          return new MkAnswerBodyMatcher(matcher);
55      }
56  
57      /**
58       * Matches the value of the MkAnswer's body bytes against the given
59       * matcher.
60       * @param matcher The matcher to use
61       * @return Matcher for checking the body of MkAnswer
62       */
63      public static Matcher<MkAnswer> hasBodyBytes(
64          final Matcher<byte[]> matcher) {
65          return new MkAnswerBodyBytesMatcher(matcher);
66      }
67  
68      /**
69       * Matches the content of the MkAnswer's header against the given
70       * matcher. Note that for a valid match to occur, the header entry must
71       * exist <i>and</i> its value(s) must match the given matcher.
72       *
73       * @param header The header to check.
74       * @param matcher The matcher to use.
75       * @return Matcher for checking the body of MkAnswer
76       */
77      public static Matcher<MkAnswer> hasHeader(final String header,
78          final Matcher<Iterable<? extends String>> matcher) {
79          return new MkAnswerHeaderMatcher(header, matcher);
80      }
81  
82  }
