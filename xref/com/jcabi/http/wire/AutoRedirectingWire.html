30  package com.jcabi.http.wire;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.http.Request;
34  import com.jcabi.http.Response;
35  import com.jcabi.http.Wire;
36  import jakarta.ws.rs.core.HttpHeaders;
37  import java.io.IOException;
38  import java.io.InputStream;
39  import java.net.HttpURLConnection;
40  import java.net.URI;
41  import java.util.Collection;
42  import java.util.List;
43  import java.util.Map;
44  import java.util.concurrent.TimeUnit;
45  import lombok.EqualsAndHashCode;
46  import lombok.ToString;
47  
48  /**
49   * Auto Redirecting Wire.
50   *
51   * <p>This wire will retry a request a certain number of times (default: 5)
52   * after a short delay when a HTTP response with a status code of 300-399 is
53   * received. On every next attempt a new URL will be used, according
54   * to the value of {@code Location} HTTP header of the response.
55   *
56   * <p>If the maximum number of retries are reached, the last response
57   * received is returned to the caller, regardless of its status code.
58   *
59   * <pre> String html = new JdkRequest("http://goggle.com")
60   *   .through(AutoRedirectingWire.class)
61   *   .header(HttpHeaders.ACCEPT, MediaType.TEXT_PLAIN)
62   *   .fetch()
63   *   .body();</pre>
64   *
65   * <p>The class is immutable and thread-safe.
66   *
67   * @since 1.6
68   */
69  @Immutable
70  @ToString(of = "origin")
71  @EqualsAndHashCode(of = { "origin", "max" })
72  public final class AutoRedirectingWire implements Wire {
73      /**
74       * Original wire.
75       */
76      private final transient Wire origin;
77  
78      /**
79       * Maximum number of retries to be made.
80       */
81      private final transient int max;
82  
83      /**
84       * Public ctor.
85       * @param wire Original wire
86       */
87      public AutoRedirectingWire(final Wire wire) {
88          this(wire, 5);
89      }
90  
91      /**
92       * Public ctor.
93       * @param wire Original wire
94       * @param retries Maximum number of retries
95       */
96      public AutoRedirectingWire(final Wire wire, final int retries) {
97          this.origin = wire;
98          this.max = retries;
99      }
100 
101     // @checkstyle ParameterNumber (5 lines)
102     @Override
103     public Response send(final Request req, final String home,
104         final String method,
105         final Collection<Map.Entry<String, String>> headers,
106         final InputStream content,
107         final int connect,
108         final int read) throws IOException {
109         Response response = this.origin.send(
110             req, home, method, headers, content, connect, read
111         );
112         int attempt = 1;
113         final URI uri = URI.create(home);
114         while (attempt < this.max) {
115             if (response.status() < HttpURLConnection.HTTP_MULT_CHOICE
116                 || response.status() >= HttpURLConnection.HTTP_BAD_REQUEST) {
117                 break;
118             }
119             final List<String> locations = response.headers().get(
120                 HttpHeaders.LOCATION
121             );
122             if (locations == null || locations.size() != 1) {
123                 break;
124             }
125             URI location = URI.create(locations.get(0));
126             if (!location.isAbsolute()) {
127                 location = uri.resolve(location);
128             }
129             response = this.origin.send(
130                 req, location.toString(),
131                 method, headers, content, connect, read
132             );
133             try {
134                 TimeUnit.SECONDS.sleep((long) attempt);
135             } catch (final InterruptedException ex) {
136                 throw new IOException(ex);
137             }
138             ++attempt;
139         }
140         return response;
141     }

143 }
