62  /**
63   * Cache for FcWire.
64   *
65   * @author Yegor Bugayenko (yegor@tpc2.com)
66   * @version $Id: ec2c4774677a912ae40133f4e3908660fce22372 $
67   * @since 1.16
68   * @checkstyle MultipleStringLiteralsCheck (500 lines)
69   */
70  @Immutable
71  @ToString
72  @EqualsAndHashCode
73  final class FcCache {
74  
75      /**
76       * Directory to keep files in.
77       */
78      private final transient String dir;
79  
80      /**
81       * Ctor.
82       */
83      FcCache() {
84          this(
85              new File(
86                  new File(System.getProperty("java.io.tmpdir")),
87                  String.format(
88                      "%s-%d",
89                      FcCache.class.getCanonicalName(),
90                      System.nanoTime()
91                  )
92              ).getAbsolutePath()
93          );
94      }
95  
96      /**
97       * Ctor.
98       * @param path Dir with files
99       */
100     FcCache(final String path) {
101         this.dir = path;
102     }
103 
104     /**
105      * Invalidate all.
106      * @throws IOException If fails
107      */
108     public void invalidate() throws IOException {
109         final File file = this.file("").getParentFile();
110         if (file.exists()) {
111             FileUtils.deleteDirectory(file);
112             Logger.debug(this, "cache invalidated in %s", file);
113         }
114     }
115 
116     /**
117      * Get and cache.
118      * @param label Label to use
119      * @param wire Original wire
120      * @param request The request
121      * @param home URI to fetch
122      * @param method HTTP method
123      * @param headers Headers
124      * @param input Input body
125      * @param connect Connect timeout
126      * @param read Read timeout
127      * @return Response
128      * @throws IOException If fails
129      * @checkstyle ParameterNumberCheck (10 lines)
130      */
131     public Response get(final String label, final Wire wire,
132         final Request request, final String home, final String method,
133         final Collection<Map.Entry<String, String>> headers,
134         final InputStream input, final int connect, final int read)
135         throws IOException {
136         final File file = this.file(label);
137         final Response rsp;
138         if (file.exists()) {
139             rsp = this.response(request, file);
140         } else {
141             rsp = this.saved(
142                 wire.send(
143                     request, home, method,
144                     headers, input, connect, read
145                 ),
146                 file
147             );
148         }
149         return rsp;
150     }
151 
152     /**
153      * Get response from file.
154      * @param req Request
155      * @param file File to read
156      * @return Response
157      * @throws IOException If fails
158      */
159     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
160     private Response response(final Request req, final File file)
161         throws IOException {
162         final JsonObject json = Json.createReader(
163             new ByteArrayInputStream(
164                 FileUtils.readFileToByteArray(file)
165             )
166         ).readObject();
167         final List<Map.Entry<String, String>> map = new LinkedList<>();
168         final JsonObject headers = json.getJsonObject("headers");
169         for (final String name : headers.keySet()) {
170             for (final JsonString value
171                 : headers.getJsonArray(name).getValuesAs(JsonString.class)) {
172                 map.add(new AbstractMap.SimpleEntry<>(name, value.getString()));
173             }
174         }
175         Logger.debug(this, "cache loaded from %s", file);
176         return new DefaultResponse(
177             req,
178             json.getInt("status"),
179             json.getString("reason"),
180             new Array<>(map),
181             json.getString("body").getBytes("UTF-8")
182         );
183     }
184 
185     /**
186      * Save response to file.
187      * @param response Response to save
188      * @param file File to read
189      * @return Response
190      * @throws IOException If fails
191      */
192     private Response saved(final Response response, final File file)
193         throws IOException {
194         final JsonObjectBuilder json = Json.createObjectBuilder();
195         json.add("status", response.status());
196         json.add("reason", response.reason());
197         final JsonObjectBuilder headers = Json.createObjectBuilder();
198         for (final Map.Entry<String, List<String>> pair
199             : response.headers().entrySet()) {
200             final JsonArrayBuilder array = Json.createArrayBuilder();
201             for (final String value : pair.getValue()) {
202                 array.add(value);
203             }
204             headers.add(pair.getKey(), array);
205         }
206         json.add("headers", headers);
207         json.add("body", response.body());
208         if (file.getParentFile().mkdirs()) {
209             Logger.debug(this, "directory created for %s", file);
210         }
211         try (final OutputStream out = new FileOutputStream(file)) {
212             Json.createWriter(out).write(json.build());
213         }
214         Logger.debug(this, "cache saved into %s", file);
215         return response;
216     }
217 
218     /**
219      * Make file from label.
220      * @param label Label to use
221      * @return File
222      */
223     private File file(final String label) {
224         final String path;
225         try {
226             path = Joiner.on("/").join(
227                 URLEncoder.encode(label, "UTF-8")
228                     .replaceAll("_", "__")
229                     .replaceAll("\\+", "_")
230                     .replaceAll("%", "_")
231                     .split("(?<=\\G.{4})")
232             );
233         } catch (final UnsupportedEncodingException ex) {
234             throw new IllegalStateException(ex);
235         }
236         return new File(this.dir, String.format("%s.json", path));
237     }
